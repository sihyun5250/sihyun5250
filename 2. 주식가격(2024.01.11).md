function solution(prices) {
    let arr= [];
    let temp= 0;
    
    for(let i=0; i<prices.length; i++){
        for(let j=i; j<prices.length; j++){
            
            //1. 마지막 index는 0을 리턴
            if(i==prices.length-1){
                temp = 0;
            //2. i번째가 j번째보다 큰 값이 있다면 j-i 후 break;   
            }else if(prices[i]>prices[j]){
                temp = j-i;
                break;
            //3. 그렇지 않으면 j-i
            }else{
                temp=j-i;
            }
        }
        arr.push(temp);
    }
    return arr;  
}
